\newcommand
\BackgroundPic
{
	\put(0,0)
	{
		\parbox[b][\paperheight]{\paperwidth}
		{
			\vfill
			\centering
			\includegraphics[width=\paperwidth, height=\paperheight, keepaspectratio]
			{background-0}
			\vfill
		}
	}
}
